#include<bits/stdc++.h>
using namespace std;

int subarray(int arr[], int n, int sum)
{
    int curr_sum = 0;
    
    for(int i=0;i<n;i++)
    {
        curr_sum = arr[i];
        for(int j= i + 1;j<n;j++)
        { 
            curr_sum += arr[j];
            if(curr_sum == sum)
            {
                cout<<i<<j<<endl;
                return 1;
            }
            if (curr_sum > sum || j == n)
                break;
            
        }
    }
    
    return 0;
}


int main()
{
    int n;cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
        cin>>arr[i];
    
    int sum = 23;
    if(subarray(arr, n, sum))
        cout<<"yes";
    else
        cout<<"no";
}
